        scene.add(gridHelper);

        const labels = new Map();

        function createLabel(text, color = '#ffffff') {
            const label = document.createElement('div');
            label.style.position = 'absolute';
            label.style.color = color;
            label.style.fontSize = '12px';
            label.style.fontWeight = 'bold';
            label.style.backgroundColor = 'rgba(0, 0, 0, 0.7)';
            label.style.padding = '4px 8px';

        function updateLabelPosition(label, position) {
            const vector = position.clone();
            vector.project(camera);
            
            const x = (vector.x * 0.5 + 0.5) * container.clientWidth;
            const y = (vector.y * -0.5 + 0.5) * container.clientHeight;
            
            label.style.left = `${x}px`;
            label.style.top = `${y}px`;
            
            // Hide label if behind camera
            label.style.display = vector.z > 1 ? 'none' : 'block';
        }

        // --- 3D Objects ---
        const objects = { hmd: null, leftHand: null, rightHand: null, ppe: {} };
        const matHead = new THREE.MeshLambertMaterial({ color: 0x00aaff });
        const matHand = new THREE.MeshLambertMaterial({ color: 0x00ffaa });
        const matPPE = new THREE.MeshLambertMaterial({ color: 0xffaa00 });
        const matPPEAttached = new THREE.MeshLambertMaterial({ color: 0xaaaaaa });
        const geoHead = new THREE.BoxGeometry(0.2, 0.15, 0.25);
        const geoHand = new THREE.SphereGeometry(0.05, 16, 16);
        const geoPPE = new THREE.BoxGeometry(0.1, 0.1, 0.1);

        function updateObject(id, pose, type, attachedTo = "") {
            let obj = type === 'ppe' ? objects.ppe[id] : objects[id];
            if (!obj) {
                let mesh;
                if (type === 'hmd') mesh = new THREE.Mesh(geoHead, matHead);
                else if (type === 'hand') mesh = new THREE.Mesh(geoHand, matHand);
                else if (type === 'ppe') mesh = new THREE.Mesh(geoPPE, matPPE);
                scene.add(mesh);
                obj = mesh;
                if (type === 'ppe') objects.ppe[id] = obj;
                else objects[id] = obj;

                // Create label for this object
                let labelText = '';
                let labelColor = '#ffffff';
                
                if (type === 'hmd') {
                    labelText = 'HMD';
                    labelColor = '#00aaff';
                } else if (type === 'hand') {
                    labelText = id === 'leftHand' ? 'Left Hand' : 'Right Hand';
                    labelColor = '#00ffaa';
                } else if (type === 'ppe') {
                    labelText = id.toUpperCase();
                    labelColor = '#ffaa00';
                }
                
                const label = createLabel(labelText, labelColor);
                labels.set(obj, label);
            }
            obj.position.set(pose.px, pose.py, -pose.pz);
            obj.quaternion.set(-pose.qx, pose.qy, pose.qz, -pose.qw);
            if (type === 'ppe') {
                if (attachedTo && attachedTo !== "") {
                    obj.material = matPPEAttached;
                    obj.scale.set(0.8, 0.8, 0.8);
                } else {
                    obj.material = matPPE;
                    obj.scale.set(1, 1, 1);
                }
            }
        }

        // --- UI Functions ---
        function updateScore(score) {
            currentScore = score;
            document.getElementById('score').textContent = score;
        }

        function addLog(message, type = 'info') {
            const container = document.getElementById('log-container');
            if (container.querySelector('.empty-state')) {
                container.innerHTML = '';
            }

            const logItem = document.createElement('div');
            logItem.className = `log-item ${type}`;
            
            const time = new Date().toLocaleTimeString(currentLang === 'pt' ? 'pt-BR' : 'en-US', { hour12: false });
            logItem.innerHTML = `<span class="log-time">${time}</span>${message}`;
            
            container.insertBefore(logItem, container.firstChild);
            
            // Keep only last 50 logs
            while (container.children.length > 50) {
                container.removeChild(container.lastChild);
            }
        }

        }

        }

        function viewFullReport() {
            if (!sessionReport) return;
            
            const modal = document.getElementById('report-modal');
            const modalBody = document.getElementById('modal-body');
            
            const summary = sessionReport.summary || {};
            const entries = sessionReport.entries || [];
            
            const minutes = Math.floor(summary.totalElapsedTime / 60);
            const seconds = Math.floor(summary.totalElapsedTime % 60);
            
            let html = `
                <div class="session-summary-grid">
                    <div class="summary-card">
                        <div class="summary-card-value">${minutes}:${seconds.toString().padStart(2, '0')}</div>
                        <div class="summary-card-label">${t('totalDuration')}</div>
                    </div>
                    <div class="summary-card">
                        <div class="summary-card-value">${summary.totalScore || 0}</div>
                        <div class="summary-card-label">${t('finalScore')}</div>
                    </div>
                    <div class="summary-card">
                        <div class="summary-card-value">${summary.tasksCompleted || 0}</div>
                        <div class="summary-card-label">${t('tasksCompleted')}</div>
                    </div>
                    <div class="summary-card">
                        <div class="summary-card-value">${summary.totalTasks || 0}</div>
                        <div class="summary-card-label">${t('totalTasks')}</div>
                    </div>
                </div>
                
                <h3 style="margin-bottom: 15px; color: #64c8ff;">${t('eventTimeline')}</h3>
                <div class="timeline">
            `;
            
            entries.forEach(entry => {
                const time = new Date(entry.timestamp).toLocaleTimeString(currentLang === 'pt' ? 'pt-BR' : 'en-US', { hour12: false });
                const eventClass = entry.eventName.toLowerCase().includes('violation') ? 'violation' : '';
                html += `
                    <div class="timeline-item ${eventClass}">
                        <div class="timeline-time">${time}</div>

                `;
            });
            
            html += '</div>';
            
            modalBody.innerHTML = html;
            modal.classList.add('open');
        }

        function downloadReport() {
            if (!sessionReport) return;
            
            const dataStr = JSON.stringify(sessionReport, null, 2);
            const dataBlob = new Blob([dataStr], { type: 'application/json' });
            const url = URL.createObjectURL(dataBlob);
            const link = document.createElement('a');
            link.href = url;
            link.download = `session_report_${new Date().toISOString().replace(/[:.]/g, '-')}.json`;
            link.click();
            URL.revokeObjectURL(url);
        }

            document.getElementById('report-modal').classList.remove('open');

        // --- WebSocket ---
        let _wsReconnecting = false;

        function connect() {
            const ws = new WebSocket(WS_URL);
            const statusIndicator = document.getElementById('status-indicator');
            const connectionStatus = document.getElementById('connection-status');

            ws.onopen = () => {
                statusIndicator.classList.add('connected');
                connectionStatus.textContent = t('connected');
                addLog(t('logConnected'), 'success');
                _wsReconnecting = false;
            };

            ws.onclose = () => {
                statusIndicator.classList.remove('connected');
                connectionStatus.textContent = t('disconnected');
                if (!_wsReconnecting) {
                    addLog(t('logDisconnected'), 'violation');
                    _wsReconnecting = true;
                }
                setTimeout(connect, 3000);
            };

            ws.onerror = (err) => console.error("WS Error", err);

            ws.onmessage = (event) => {
                try {
                    const envelope = JSON.parse(event.data);
                    handleEvent(envelope);
                } catch (e) {
                    console.error("Parse error:", e);
                }
            };
        }

        function handleEvent(envelope) {
            const type = envelope.eventType;
            const payload = envelope.payload;

            switch (type) {
                case "PoseFrame":
                    if (payload.hmd) updateObject('hmd', payload.hmd, 'hmd');
                    if (payload.leftHand) updateObject('leftHand', payload.leftHand, 'hand');
                    if (payload.rightHand) updateObject('rightHand', payload.rightHand, 'hand');
                    if (payload.ppe && Array.isArray(payload.ppe)) {
                        payload.ppe.forEach(p => updateObject(p.id, p.pose, 'ppe', p.attachedTo));
                    }
                    break;

                case "SessionStarted":
                    document.getElementById('session-id').textContent = 
                        `Session: ${payload.sessionId.substring(0, 8)}`;
                    addLog('🎮 Training session started', 'success');
                    break;

                case "SessionManifest":
                    if (payload.tasks) {
                        payload.tasks.forEach(task => {
                            addOrUpdateTask(task.taskName, task.taskName, task.description, 'pending');
                        });
                    }
                    break;

                case "TaskStarted":
                    addOrUpdateTask(payload.taskId, payload.taskName, payload.taskDescription, 'active');
                    addLog(`▶️ Started: ${payload.taskName}`);
                    break;

                case "TaskCompleted":
                    addOrUpdateTask(payload.taskId, payload.taskName, payload.taskDescription, 'completed');
                    addLog(`✅ Completed: ${payload.taskName}`, 'success');
                    break;

                case "TaskTimeout":
                    addOrUpdateTask(payload.taskId, payload.taskName, payload.taskDescription, 'failed');
                    addLog(`⏰ Timeout: ${payload.taskName}`, 'violation');
                    break;

                case "ScoreChanged":
                    const delta = payload.delta;
                    updateScore(payload.totalScore);
                    if (delta > 0) {
                        addLog(`➕ Score +${delta} (Total: ${payload.totalScore})`, 'success');
                    } else if (delta < 0) {
                        addLog(`➖ Score ${delta} (Total: ${payload.totalScore})`, 'violation');
                    }
                    break;

                case "SafetyViolation":
                    addLog(`⚠️ Safety Violation: ${payload.message}`, 'violation');
                    break;

                case "CriticalSafetyFailure":
                    addLog(`🚨 CRITICAL: ${payload.reason}`, 'violation');
                    break;

                case "SessionCompleted":
                    addLog(`🏁 Session Complete! Score: ${payload.totalScore}, Tasks: ${payload.tasksCompleted}/${payload.totalTasks}`, 'success');
                    break;

                case "SessionLogFile":
                    if (payload.content) {
                        try {
                            const report = JSON.parse(payload.content);
                            updateReportSection(report);
                            addLog('📊 Session report available', 'success');
                        } catch (e) {
                            console.error('Failed to parse session log:', e);
                        }
                    }
                    break;

                case "GroupStarted":
                    addLog(`📂 Group Started: ${payload.groupName}`);
                    break;

                case "GroupCompleted":
                    addLog(`📁 Group Completed: ${payload.groupName}`, 'success');
                    break;

                case "SessionReset":
                    // Clear all UI state
                    tasks.clear();
                    sessionReport = null;
                    currentScore = 0;
                    
                    // Reset UI elements
                    updateScore(0);
                    document.getElementById('task-list').innerHTML = `
                        <div class="empty-state">
                            <div class="empty-state-icon">⏳</div>
                            <div>Waiting for session to start...</div>
                        </div>
                    `;
                    document.getElementById('report-section').innerHTML = `
                        <div class="empty-state">
                            <div class="empty-state-icon">📄</div>
                            <div>Report available after session</div>
                        </div>
                    `;
                    document.getElementById('session-id').textContent = 'No Session';
                    
                    addLog('🔄 Session has been reset', 'info');
                    break;
            }
        }
