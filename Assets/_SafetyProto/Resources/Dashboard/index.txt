    /* ================================================================
     * WsManager — exponential backoff, jitter, keepalive ping
     * ============================================================== */
    class WsManager {
      constructor(url, onMessage) {
        this._url       = url;
        this._onMessage = onMessage;
        this._ws        = null;
        this._retries   = 0;
        this._retryTimer= null;
        this._pingTimer = null;
        this._intentional = false;
      }

      connect() {
        this._intentional = false;
        this._attempt();
      }

      _attempt() {
        clearTimeout(this._retryTimer);
        setStatusUI('connecting');

        try {
          this._ws = new WebSocket(this._url);
        } catch {
          this._scheduleRetry();
          return;
        }

        this._ws.onopen = () => {
          this._retries = 0;
          setStatusUI('connected');
          addLog(t('logConnected'), 'success');

          // NOVO: Pedir sincronização de estado (evita amnésia após queda de conexão)
          try {
            this._ws.send(JSON.stringify({ eventType: "RequestSync" }));
          } catch (e) {
            console.error("Erro ao solicitar sync:", e);
          }

          // keepalive ping every 8 s
          this._pingTimer = setInterval(() => {
            if (this._ws?.readyState === WebSocket.OPEN) {
              try { this._ws.send('ping'); } catch (_) {}
            }
          }, 8_000);
        };

        this._ws.onmessage = ev => {
          try { this._onMessage(JSON.parse(ev.data)); }
          catch { /* ignore malformed */ }
        };

        this._ws.onerror = () => { /* handled in onclose */ };

        this._ws.onclose = () => {
          clearInterval(this._pingTimer);
          if (this._intentional) return;
          setStatusUI('disconnected');
          addLog(t('logDisconnected'), 'violation');
          this._scheduleRetry();
        };
      }

      _scheduleRetry() {
        const base = Math.min(RECONNECT_BASE_MS * (2 ** this._retries), RECONNECT_MAX_MS);
        const delay = base * (1 + (Math.random() * 2 - 1) * RECONNECT_JITTER);
        this._retries++;
        this._retryTimer = setTimeout(() => this._attempt(), delay);
      }

      destroy() {
        this._intentional = true;
        clearTimeout(this._retryTimer);
        clearInterval(this._pingTimer);
        this._ws?.close();
      }
    }

    /* ================================================================
     * EVENT ROUTER — declarative handler map
     * ============================================================== */
    const handlers = {};
    const on = (type, fn) => { handlers[type] = fn; };

    function route(envelope) {
      const { eventType, payload } = envelope;
      if (!eventType) return;
      const fn = handlers[eventType];
      if (fn) { try { fn(payload); } catch (e) { console.error(`[handler:${eventType}]`, e); } }
      // PoseFrame handled inline (high frequency — skip addLog)
    }

    /* ── Task events ── */
    function upsertTask(p, status) {
      const existing = tasks.get(p.taskId) ?? {};
      tasks.set(p.taskId, {
        id:            p.taskId,
        name:          p.taskName          ?? existing.name          ?? p.taskId,
        description:   p.taskDescription   ?? existing.description   ?? '',
        groupName:     p.groupName         ?? existing.groupName     ?? '',
        order:         p.order             ?? existing.order         ?? 0,
        executionMode: p.executionMode     ?? existing.executionMode ?? '',
        expectedAction:p.expectedAction    ?? existing.expectedAction?? '',
        requiredPpe:   p.requiredPpe       ?? existing.requiredPpe   ?? [],
        successPoints: p.successPoints     ?? existing.successPoints ?? 0,
        failurePenalty:p.failurePenalty    ?? existing.failurePenalty?? 0,
        ppePenalty:    p.ppePenalty        ?? existing.ppePenalty    ?? 0,
        status,
      });
      renderTasks();
    }

    on('TaskStarted',   p => { upsertTask(p, 'active');    addLog(`${t('logStarted')}: ${p.taskName}`); });
    on('TaskCompleted', p => { upsertTask(p, 'completed'); addLog(`${t('logCompleted')}: ${p.taskName}`, 'success'); });
    on('TaskTimeout',   p => { upsertTask(p, 'failed');    addLog(`${t('logTimeout')}: ${p.taskName}`, 'violation'); });

    /* ── Task renderer — full rebuild from Map ── */
    function renderTasks() {
      const container = document.getElementById('task-list');
      if (tasks.size === 0) { resetTaskUI(); return; }

      // Sort: active first, then by order
      let sorted = [...tasks.values()];

      if (hideCompleted) {
        sorted = sorted.filter(t => t.status !== 'completed' && t.status !== 'failed');
      }

      sorted.sort((a, b) => {
        const priority = s => s === 'active' ? 0 : s === 'pending' ? 1 : 2;
        const pd = priority(a.status) - priority(b.status);
        if (pd !== 0) return pd;
        return (a.order ?? 999) - (b.order ?? 999);
      });

      container.innerHTML = '';
      sorted.forEach(task => {
        const pts = [];
        if (task.successPoints)  pts.push(`+${task.successPoints}pts`);
        if (task.ppePenalty)     pts.push(`-${task.ppePenalty} EPI`);
        if (task.failurePenalty) pts.push(`-${task.failurePenalty} falha`);

        const ppe = (task.requiredPpe ?? []).join(', ');

        const el = document.createElement('div');
        el.className = `task-item ${task.status}`;
        el.innerHTML = `
          <div class="task-header">
            <div class="task-name">${task.name}</div>
            <div class="task-badge ${task.status}">${t(task.status)}</div>
          </div>
          ${task.description ? `<div class="task-description">${task.description}</div>` : ''}
          <div class="task-meta">
            ${task.groupName ? `<span>Grupo: ${task.groupName}</span>` : ''}
            ${pts.length ? `<span class="task-pts">${pts.join(' / ')}</span>` : ''}
            ${ppe ? `<span>EPI: ${ppe}</span>` : ''}
          </div>
        `;
        container.appendChild(el);
      });
    }

    /* ================================================================
     * EvaluatorViewport — encapsulates Three.js scene, object pool,
     * label system and render loop. Completely decoupled from UI state.
     * ============================================================== */
    class EvaluatorViewport {
      constructor(containerEl) {
        this._container = containerEl;

        /* Scene */
        this._scene = new THREE.Scene();
        this._scene.background = new THREE.Color(0x0a0e27);
        this._scene.fog = new THREE.Fog(0x0a0e27, 6, 30);

        /* Camera */
        this._camera = new THREE.PerspectiveCamera(
          75,
          containerEl.clientWidth / containerEl.clientHeight,
          0.1, 1000
        );
        this._camera.position.set(0, 1.6, 3);

        /* Renderer */
        this._renderer = new THREE.WebGLRenderer({ antialias: true });
        this._renderer.setPixelRatio(window.devicePixelRatio);
        this._renderer.setSize(containerEl.clientWidth, containerEl.clientHeight);
        containerEl.appendChild(this._renderer.domElement);

        /* Controls */
        this._controls = new OrbitControls(this._camera, this._renderer.domElement);
        this._controls.target.set(0, 1.0, 0);
        this._controls.update();

        /* Lighting */
        this._scene.add(new THREE.AmbientLight(0x404040, 2));
        const dir = new THREE.DirectionalLight(0xffffff, 1);
        dir.position.set(2, 5, 2);
        this._scene.add(dir);

        /* Environment helpers */
        this._scene.add(new THREE.GridHelper(10, 10, 0x333355, 0x222244));
        this._scene.add(new THREE.AxesHelper(1));

        /* Label overlay */
        this._labelOverlay = document.createElement('div');
        Object.assign(this._labelOverlay.style, {
          position: 'absolute', top: '0', left: '0',
          width: '100%', height: '100%', pointerEvents: 'none',
        });
        containerEl.appendChild(this._labelOverlay);

        /* Object + label pools */
        this._objects   = { hmd: null, leftHand: null, rightHand: null, ppe: {} };
        this._labels    = new Map();   // mesh → { el, yOffset }
        this._ppeColors = new Map();   // ppeId → { css, mat }
        this._ppeColorIdx = 0;

        /* Shared geometries */
        this._geoHead = new THREE.BoxGeometry(0.2, 0.15, 0.25);
        this._geoHand = new THREE.SphereGeometry(0.05, 16, 16);
        this._geoPPE  = new THREE.BoxGeometry(0.1, 0.1, 0.1);

        /* Shared materials */
        this._matHead       = new THREE.MeshLambertMaterial({ color: 0x00aaff });
        this._matHand       = new THREE.MeshLambertMaterial({ color: 0x00ffaa });
        this._matPPEAttached= new THREE.MeshLambertMaterial({ color: 0xaaaaaa });

        this._PPE_PALETTE = [
          { hex: 0xff6b35, css: '#ff6b35' },
          { hex: 0xc44dff, css: '#c44dff' },
          { hex: 0x00cfff, css: '#00cfff' },
          { hex: 0xff4da6, css: '#ff4da6' },
          { hex: 0x44ff88, css: '#44ff88' },
        ];

        /* Resize listener */
        this._onResize = () => this.resize();
        window.addEventListener('resize', this._onResize);

        /* Start loop */
        this._animate();
      }

      /* ── Public API ───────────────────────────────────────────── */

      /** Call after the container element changes size (e.g. sidebar resize). */
      resize() {
        const w = this._container.clientWidth;
        const h = this._container.clientHeight;
        this._camera.aspect = w / h;
        this._camera.updateProjectionMatrix();
        this._renderer.setSize(w, h);
      }

      /** Update pose for hmd / hand / ppe. Called at 10 Hz from PoseFrame. */
      updatePose(id, pose, type, attachedTo = '') {
        let obj = type === 'ppe' ? this._objects.ppe[id] : this._objects[id];

        if (!obj) {
          obj = this._createObject(id, type);
        }

        obj.position.set(pose.px, pose.py, -pose.pz);
        obj.quaternion.set(-pose.qx, pose.qy, pose.qz, -pose.qw);

        if (type === 'ppe') {
          if (attachedTo) {
            obj.material = this._matPPEAttached;
            obj.scale.setScalar(0.8);
          } else {
            obj.material = this._ppeColors.get(id)?.mat ?? this._matPPEAttached;
            obj.scale.setScalar(1);
          }
        }
      }

      /** Release GPU resources (call if viewport is ever torn down). */
      dispose() {
        window.removeEventListener('resize', this._onResize);
        this._renderer.dispose();
      }

      /* ── Private helpers ──────────────────────────────────────── */

      _createObject(id, type) {
        let mesh;
        if (type === 'hmd') {
          mesh = new THREE.Mesh(this._geoHead, this._matHead);
        } else if (type === 'hand') {
          mesh = new THREE.Mesh(this._geoHand, this._matHand);
        } else {
          const p = this._PPE_PALETTE[this._ppeColorIdx % this._PPE_PALETTE.length];
          const mat = new THREE.MeshLambertMaterial({ color: p.hex });
          this._ppeColors.set(id, { css: p.css, mat });
          this._ppeColorIdx++;
          mesh = new THREE.Mesh(this._geoPPE, mat);
        }

        this._scene.add(mesh);
        if (type === 'ppe') this._objects.ppe[id] = mesh;
        else                this._objects[id] = mesh;

        /* Create matching label */
        let text = '', color = '#ffffff', yOffset = 0.15;
        if      (type === 'hmd')  { text = 'HMD';  color = '#00aaff'; yOffset = 0.28; }
        else if (type === 'hand') { text = id === 'leftHand' ? 'Left Hand' : 'Right Hand';
                                    color = '#00ffaa'; yOffset = 0.12; }
        else                      { text = id.toUpperCase();
                                    color = this._ppeColors.get(id)?.css ?? '#ffaa00'; yOffset = 0.14; }
        this._labels.set(mesh, this._createLabel(text, color, yOffset));

        return mesh;
      }

      _createLabel(text, color, yOffset) {
        const el = document.createElement('div');
        Object.assign(el.style, {
          position: 'absolute', color,
          fontSize: '12px', fontWeight: 'bold',
          backgroundColor: 'rgba(0,0,0,0.7)',
          padding: '3px 8px', borderRadius: '4px',
          border: `1px solid ${color}`,
          whiteSpace: 'nowrap',
          transform: 'translate(-50%,-100%)',
          marginTop: '-10px',
        });
        el.textContent = text;
        this._labelOverlay.appendChild(el);
        return { el, yOffset };
      }

      _projectLabel(ld, worldPos) {
        const v = worldPos.clone().project(this._camera);
        const x = (v.x * 0.5 + 0.5) * this._container.clientWidth;
        const y = (v.y * -0.5 + 0.5) * this._container.clientHeight;
        const visible = v.z <= 1;
        ld.el.style.display = visible ? 'block' : 'none';
        ld.el.style.left = `${x}px`;
        ld.el.style.top  = `${y}px`;
        return { x, y, visible };
      }

      _animate() {
        requestAnimationFrame(() => this._animate());
        this._controls.update();
        this._updateLabels();
        this._renderer.render(this._scene, this._camera);
      }

      _updateLabels() {
        const visible = [];
        this._labels.forEach((ld, obj) => {
          if (!obj?.position) return;
          const wp = obj.position.clone();
          wp.y += ld.yOffset;
          const sp = this._projectLabel(ld, wp);
          if (sp.visible) visible.push({ el: ld.el, x: sp.x, y: sp.y });
        });

        /* Iterative screen-space push-apart (4 passes, 96×22 px label estimate) */
        const LW = 96, LH = 22, PAD = 6;
        for (let iter = 0; iter < 4; iter++) {
          for (let i = 0; i < visible.length; i++) {
            for (let j = i + 1; j < visible.length; j++) {
              const a = visible[i], b = visible[j];
              const ox = (LW + PAD) - Math.abs(b.x - a.x);
              const oy = (LH + PAD) - Math.abs(b.y - a.y);
              if (ox > 0 && oy > 0) {
                if (ox < oy) {
                  const push = ox / 2 * Math.sign(b.x - a.x || 1);
                  a.x -= push; b.x += push;
                } else {
                  const push = oy / 2 * Math.sign(b.y - a.y || -1);
                  a.y -= push; b.y += push;
                }
              }
            }
          }
        }
        visible.forEach(({ el, x, y }) => {
          el.style.left = `${x}px`;
          el.style.top  = `${y}px`;
        });
      }
    }

    /* ================================================================
     * SidebarResizeManager — drag-to-resize with localStorage persist
     * ============================================================== */
    class SidebarResizeManager {
      /**
       * @param {HTMLElement} sidebar   — the element whose width changes
       * @param {HTMLElement} handle    — the drag target on the right edge
       * @param {EvaluatorViewport} viewport — notified after each resize
       * @param {{ min: number, max: number, default: number }} opts
       */
      constructor(sidebar, handle, viewport, opts = {}) {
        this._sidebar  = sidebar;
        this._handle   = handle;
        this._viewport = viewport;
        this._min      = opts.min     ?? 260;
        this._max      = opts.max     ?? 700;
        this._default  = opts.default ?? 380;
        this._storageKey = 'sp_sidebar_w';

        this._dragging  = false;
        this._startX    = 0;
        this._startW    = 0;

        this._restore();
        this._bind();
      }

      /* ── Private ── */

      _restore() {
        const saved = parseInt(localStorage.getItem(this._storageKey), 10);
        const w = (!isNaN(saved) && saved >= this._min && saved <= this._max)
          ? saved : this._default;
        this._applyWidth(w);
      }

      _applyWidth(w) {
        const clamped = Math.max(this._min, Math.min(this._max, w));
        this._sidebar.style.setProperty('--sidebar-w', `${clamped}px`);
        // Override the CSS var directly on the element for immediate effect
        this._sidebar.style.width = `${clamped}px`;
      }

      _bind() {
        this._handle.addEventListener('mousedown', e => this._onMouseDown(e));
        /* Touch support */
        this._handle.addEventListener('touchstart', e => this._onTouchStart(e), { passive: false });
      }

      _onMouseDown(e) {
        e.preventDefault();
        this._dragging = true;
        this._startX   = e.clientX;
        this._startW   = this._sidebar.offsetWidth;
        this._handle.classList.add('dragging');
        document.body.classList.add('resizing');

        const onMove = e => this._onMouseMove(e);
        const onUp   = () => {
          this._dragging = false;
          this._handle.classList.remove('dragging');
          document.body.classList.remove('resizing');
          this._save();
          document.removeEventListener('mousemove', onMove);
          document.removeEventListener('mouseup', onUp);
          /* Tell the viewport its container changed size */
          this._viewport?.resize();
        };

        document.addEventListener('mousemove', onMove);
        document.addEventListener('mouseup', onUp);
      }

      _onMouseMove(e) {
        if (!this._dragging) return;
        const delta = e.clientX - this._startX;
        this._applyWidth(this._startW + delta);
      }

      _onTouchStart(e) {
        e.preventDefault();
        const touch = e.touches[0];
        this._dragging = true;
        this._startX   = touch.clientX;
        this._startW   = this._sidebar.offsetWidth;
        this._handle.classList.add('dragging');

        const onMove = e => {
          const t = e.touches[0];
          const delta = t.clientX - this._startX;
          this._applyWidth(this._startW + delta);
        };
        const onEnd = () => {
          this._dragging = false;
          this._handle.classList.remove('dragging');
          this._save();
          this._viewport?.resize();
          document.removeEventListener('touchmove', onMove);
          document.removeEventListener('touchend', onEnd);
        };

        document.addEventListener('touchmove', onMove, { passive: false });
        document.addEventListener('touchend', onEnd);
      }

      _save() {
        localStorage.setItem(this._storageKey, this._sidebar.offsetWidth);
      }
    }

    /* ================================================================
     * MAIN ENTRY POINT
     * ============================================================== */
    const canvasContainer = document.getElementById('three-canvas');

    /* Boot 3D viewport */
    const viewport = new EvaluatorViewport(canvasContainer);

    /* PoseFrame handler — bypasses router (high-frequency) */
    function handlePoseFrame(p) {
      if (p.hmd)       viewport.updatePose('hmd',       p.hmd,       'hmd');
      if (p.leftHand)  viewport.updatePose('leftHand',  p.leftHand,  'hand');
      if (p.rightHand) viewport.updatePose('rightHand', p.rightHand, 'hand');
      if (Array.isArray(p.ppe)) {
        p.ppe.forEach(item => viewport.updatePose(item.id, item.pose, 'ppe', item.attachedTo));
      }
    }

    /* Boot sidebar resize manager (needs viewport reference for resize callback) */
    new SidebarResizeManager(
      document.getElementById('sidebar'),
      document.getElementById('resize-handle'),
      viewport,
      { min: 260, max: 700, default: 380 }
    );

    /* Boot WebSocket */
    const ws = new WsManager(
      `ws://${window.location.hostname}:${WS_PORT}${WS_PATH}`,
      envelope => {
        if (envelope.eventType === 'PoseFrame') {
          handlePoseFrame(envelope.payload);
        } else {
          route(envelope);
        }
      }
    );

    ws.connect();
    applyTranslations();

  </script>
