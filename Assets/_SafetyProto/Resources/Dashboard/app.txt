import * as THREE from 'three';
import { OrbitControls } from 'three/addons/controls/OrbitControls.js';

// --- Configuration ---
const WS_PORT = 7071; // Must match EvaluatorDashboardBootstrap.cs
const WS_URL = `ws://${window.location.hostname}:${WS_PORT}/eval`;

// --- Scene Setup ---
const scene = new THREE.Scene();
scene.background = new THREE.Color(0x222222);
scene.fog = new THREE.Fog(0x222222, 5, 30);

// Camera
const camera = new THREE.PerspectiveCamera(75, window.innerWidth / window.innerHeight, 0.1, 1000);
camera.position.set(0, 1.6, 3); // Positioned behind/above typical player height

// Renderer
const renderer = new THREE.WebGLRenderer({ antialias: true });
renderer.setSize(window.innerWidth, window.innerHeight);
document.body.appendChild(renderer.domElement);

// Controls
const controls = new OrbitControls(camera, renderer.domElement);
controls.target.set(0, 1.0, 0);
controls.update();

// Lighting
const ambientLight = new THREE.AmbientLight(0x404040, 2); // Soft white light
scene.add(ambientLight);
const dirLight = new THREE.DirectionalLight(0xffffff, 1);
dirLight.position.set(2, 5, 2);
scene.add(dirLight);

// Floor Grid
const gridHelper = new THREE.GridHelper(10, 10, 0x444444, 0x444444);
scene.add(gridHelper);

// Axis Helper (Red=X, Green=Y, Blue=Z)
const axesHelper = new THREE.AxesHelper(1);
scene.add(axesHelper);

// --- Objects Map ---
// We will store references to our 3D objects here to update them by ID
const objects = {
    hmd: null,
    leftHand: null,
    rightHand: null,
    ppe: {} // Keyed by PPE ID
};

// --- Materials & Geometries ---
const matHead = new THREE.MeshLambertMaterial({ color: 0x00aaff });
const matHand = new THREE.MeshLambertMaterial({ color: 0x00ffaa });
const matPPE = new THREE.MeshLambertMaterial({ color: 0xffaa00 });
const matPPEAttached = new THREE.MeshLambertMaterial({ color: 0xaaaaaa }); // Gray when attached (worn)

const geoHead = new THREE.BoxGeometry(0.2, 0.15, 0.25);
const geoHand = new THREE.SphereGeometry(0.05, 16, 16);
const geoPPE = new THREE.BoxGeometry(0.1, 0.1, 0.1);

// Helper to create or update an object
function updateObject(id, pose, type, attachedTo = "") {
    let obj = type === 'ppe' ? objects.ppe[id] : objects[id];

    // Create if doesn't exist
    if (!obj) {
        let mesh;
        if (type === 'hmd') mesh = new THREE.Mesh(geoHead, matHead);
        else if (type === 'hand') mesh = new THREE.Mesh(geoHand, matHand);
        else if (type === 'ppe') mesh = new THREE.Mesh(geoPPE, matPPE);
        
        scene.add(mesh);
        obj = mesh;

        // Store reference
        if (type === 'ppe') objects.ppe[id] = obj;
        else objects[id] = obj;
        
        console.log(`Created object: ${id}`);
    }

    // Update Position
    // Unity uses Y-up, LHS. Three.js uses Y-up, RHS.
    // Usually need to negate Z for position and flip specific quat components.
    // Your server sends standard floats. We negate Z to match standard WebGL view.
    obj.position.set(pose.px, pose.py, -pose.pz);

    // Update Rotation
    // Unity (LHS) -> Three.js (RHS) quaternion conversion:
    // x = -x, y = y, z = z, w = -w  (Or similar permutation depending on exact coordinate logic)
    // For simple viz, direct mapping usually looks "okay" but mirrored.
    // Correct mapping for Unity -> Three.js: (-x, y, z, -w)
    obj.quaternion.set(-pose.qx, pose.qy, pose.qz, -pose.qw);

    // Visual tweaks based on state
    if (type === 'ppe') {
        if (attachedTo && attachedTo !== "") {
            obj.material = matPPEAttached; // It's attached to something
            obj.scale.set(0.8, 0.8, 0.8);
        } else {
            obj.material = matPPE; // It's loose in the world
            obj.scale.set(1, 1, 1);
        }
    }
}

// --- WebSocket Logic ---
const statusSpan = document.getElementById('status');
const sessionSpan = document.getElementById('session-id');
const scoreSpan = document.getElementById('score');
const logDiv = document.getElementById('log');

function log(msg) {
    const el = document.createElement('div');
    el.textContent = `> ${msg}`;
    logDiv.prepend(el);
    if (logDiv.children.length > 10) logDiv.lastChild.remove();
}

function connect() {
    console.log(`Connecting to ${WS_URL}...`);
    const ws = new WebSocket(WS_URL);

    ws.onopen = () => {
        console.log("Connected!");
        statusSpan.textContent = "Connected";
        statusSpan.classList.add("connected");
        log("WebSocket Connected");
    };

    ws.onclose = () => {
        console.log("Disconnected.");
        statusSpan.textContent = "Disconnected";
        statusSpan.classList.remove("connected");
        setTimeout(connect, 3000); // Auto-reconnect
    };

    ws.onerror = (err) => {
        console.error("WS Error", err);
    };

    ws.onmessage = (event) => {
        try {
            const envelope = JSON.parse(event.data);
            handleEvent(envelope);
        } catch (e) {
            console.error("Parse error:", e);
        }
    };
}

function handleEvent(envelope) {
    const type = envelope.eventType;
    const payload = envelope.payload;

    if (type === "PoseFrame") {
        // --- 60Hz/10Hz High Frequency Data ---
        // Update HMD
        if (payload.hmd) updateObject('hmd', payload.hmd, 'hmd');
        
        // Update Hands
        if (payload.leftHand) updateObject('leftHand', payload.leftHand, 'hand');
        if (payload.rightHand) updateObject('rightHand', payload.rightHand, 'hand');

        // Update PPE
        if (payload.ppe && Array.isArray(payload.ppe)) {
            payload.ppe.forEach(p => {
                updateObject(p.id, p.pose, 'ppe', p.attachedTo);
            });
        }
    } 
    else if (type === "ScoreChanged") {
        scoreSpan.textContent = payload.totalScore;
        log(`Score: ${payload.totalScore} (${payload.delta > 0 ? '+' : ''}${payload.delta})`);
    }
    else if (type === "SessionStarted") {
        sessionSpan.textContent = payload.sessionId.substring(0, 8);
        log("Session Started");
    }
    else if (type === "ActionAttempt") {
        log(`Action: ${payload.actionType}`);
    }
    else if (type === "SafetyViolation") {
        log(`VIOLATION: ${payload.violationCode}`);
    }
    // Handle other events as needed...
}

// --- Render Loop ---
function animate() {
    requestAnimationFrame(animate);
    controls.update();
    renderer.render(scene, camera);
}

// Start
window.addEventListener('resize', () => {
    camera.aspect = window.innerWidth / window.innerHeight;
    camera.updateProjectionMatrix();
    renderer.setSize(window.innerWidth, window.innerHeight);
});

connect();
animate();
